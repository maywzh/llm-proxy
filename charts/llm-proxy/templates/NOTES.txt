1. Get the proxy URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "llm-proxy.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "llm-proxy.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "llm-proxy.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  echo "Proxy URL: http://127.0.0.1:{{ .Values.service.port }}"
{{- end }}

2. Use the proxy (OpenAI-compatible):
  curl http://<PROXY_URL>/v1/chat/completions \
    -H "Authorization: Bearer <YOUR_CREDENTIAL_KEY>" \
    -H "Content-Type: application/json" \
    -d '{"model": "gpt-4", "messages": [{"role": "user", "content": "Hello"}]}'

3. Access the Admin API:
  curl http://<PROXY_URL>/admin/v1/providers \
    -H "Authorization: Bearer <YOUR_ADMIN_KEY>"

{{- if .Values.adminUI.enabled }}

4. Admin UI:
{{- if .Values.adminUI.ingress.enabled }}
{{- range $host := .Values.adminUI.ingress.hosts }}
  http{{ if $.Values.adminUI.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "llm-proxy.fullname" . }}-admin-ui {{ .Values.adminUI.service.port }}:{{ .Values.adminUI.service.port }}
{{- end }}
{{- end }}

{{- if .Values.prometheus.enabled }}

5. Prometheus:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "llm-proxy.fullname" . }}-prometheus {{ .Values.prometheus.service.port }}:{{ .Values.prometheus.service.port }}
{{- end }}

{{- if .Values.grafana.enabled }}

6. Grafana:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "llm-proxy.fullname" . }}-grafana {{ .Values.grafana.service.port }}:{{ .Values.grafana.service.port }}
  Default credentials: admin / (see values.yaml or secret)
{{- end }}
